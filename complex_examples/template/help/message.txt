    {bold Options:}

      --ext       extension to use for default template [default: 'js']

      --withHelp  when set, generates a help action and files

  {blue.bold action add}
    add a new template file to a generator

    {bold Usage:}
      {yellow hygen} {yellow.bold template add} --name NAME --generator GENERATOR --action ACTION

    {bold Options:}
      --name       name of the generator
      --generator  name of the generator
      --action     action to create for named generator

  {cyan action help}
    show this message

    {bold Usage:}
      {yellow hygen} {yellow.bold template help}
